[docs]def get_endianness():
    import array
    end =  ord(array.array("i",[1]).tostring()[0])
    
    if end:
        # this is little endian
        return GiftiEndian.GIFTI_ENDIAN_LITTLE
    else:
        # this is big endian
        return GiftiEndian.GIFTI_ENDIAN_BIG
