[docs]class TVBException(Exception):
    """
    Base class for all TVB exceptions.
    """


    def __init__(self, message, parent_exception=None):
        Exception.__init__(self, message, parent_exception)
        self.message = message


    def __repr__(self):
        return self.message
    
    def __str__(self):
        return self.message
